## Select field

The `SelectField` component renders a select field.

<Tabs
  defaultValue="admin-simple"
  values={[
    { label: "Component - simple", value: "admin-simple" },
    { label: "Component - advanced", value: "admin-advanced" },
    { label: "API schema", value: "schema" },
  ]}
>
  <TabItem value="admin-simple">

```tsx
<SelectField 
  field="category" 
  label="Select category" 
  options="Category.name" 
/>
```

  </TabItem>

  <TabItem value="admin-advanced">

```tsx
function renderSelectFieldOptions(accessor) {
  const {value: name} = accessor.getField("name")
  const {value: color} = accessor.getField("color")

  return <span style={{ backgroundColor: color }}>{name}</span>
}

<SelectField
  field="category"
  label="Select category"
  options="Category"
  renderOption={renderSelectFieldOptions}
  optionsStaticRender={
    <>
      <Field field="name" />
      <Field field="color" />
    </>
  }
/>
```

  </TabItem>

  <TabItem value="schema">

```ts
export class Article {
  category = def.manyHasOne(Category);
}

export class Category {
  name = def.stringColumn();
  color = def.stringColumn();
}
```

  </TabItem>
</Tabs>

## Multi select field

The `MultiSelectField` renders a select field with posibility to select multiple options.

<Tabs
  defaultValue="admin"
  values={[
    { label: "Component", value: "admin" },
    { label: "API schema", value: "schema" },
  ]}
>
  <TabItem value="admin">

```tsx
<MultiSelectField field="tags" label="Select tags" options="Tag.name" />
```

  </TabItem>

  <TabItem value="schema">

```ts
export class Article {
  tags = def.manyHasMany(Tag, "articles");
}

export class Tag {
  name = def.stringColumn();
  articles = def.manyHasManyInverse(Article, "tags");
}
```

  </TabItem>
</Tabs>

## Repeater field

The `Repeater` component renders a repeater field with given child components.

<Tabs
  defaultValue="admin"
  values={[
    { label: "Component", value: "admin" },
    { label: "API schema", value: "schema" },
  ]}
>
  <TabItem value="admin">

```tsx
<Repeater field="tasks" label="Todo list" sortableBy="order">
    <TextAreaField field="content" label="Task" />
</Reapeater>
```

  </TabItem>
  <TabItem value="schema">

```ts
export class TodoList {
  tasks = def.oneHasMany(Task, "todoList");
}

export class Task {
  content = def.stringColumn();
  todoList = def.manyHasOne(TodoItem, "tasks");
}
```

  </TabItem>
</Tabs>
